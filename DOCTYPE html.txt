<body>
  <h2>Test Booking Bot Demo</h2>
  <div id="chat"></div>
  
  <div id="quickReplies">
    <button class="quick-btn" onclick="sendQuick('Check available slots')">Check available slots</button>
    <button class="quick-btn" onclick="sendQuick('Book a test')">Book a test</button>
    <button class="quick-btn" onclick="sendQuick('Contact support')">Contact support</button>
  </div>
  
  <input type="text" id="userInput" placeholder="Type your message..." />
  <button onclick="sendMessage()">Send</button>

  <script>
    const chat = document.getElementById('chat');
    const input = document.getElementById('userInput');

    // Add a new message to the chat
    function addMessage(sender, text) {
      const div = document.createElement('div');
      div.className = sender;
      div.innerText = text;
      chat.appendChild(div);
      chat.scrollTop = chat.scrollHeight;
    }

    // Handle quick reply buttons
    function sendQuick(text) {
      input.value = text;
      sendMessage();
    }

    // Handle sending messages
    function sendMessage() {
      const msg = input.value.trim();
      if(!msg) return;
      addMessage('user', msg);
      input.value = '';

      setTimeout(() => {
        const text = msg.toLowerCase();

        // Responses for earlier test slots
        if(text.includes('earlier test') || text.includes('check available slots')) {
          addMessage('bot', 'I found a few earlier test slots: Sep 5, Sep 7, Sep 10. Would you like to book one?');
        } 
        // Responses for booking
        else if(text.includes('book')) {
          addMessage('bot', 'Booking confirmed! Youâ€™ll receive a notification soon. âœ…');
        } 
        // Responses for support
        else if(text.includes('contact')) {
          addMessage('bot', 'You can reach support at support@example.com or call 123-456-7890.');
        } 
        // Handle unexpected inputs
        else {
          addMessage('bot', "I'm sorry, I didn't quite catch that. Please try asking about earlier test slots or booking a test.");
        }
      }, 500);
    }

    // Optional: Welcome message
    addMessage('bot', "Hi there! ðŸ‘‹ I'm your test booking assistant. I can help you find and book earlier test dates. Try using the quick buttons or type your request.");
  </script>
